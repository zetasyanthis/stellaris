#Find nearest planet
country_event = {
	id = fatalf.031
	hide_window = yes
	is_triggered_only = yes
	location = root
	
	immediate = {
		every_owned_fleet = {
			queue_actions = {
				repeat = {
					find_closest_planet = {
						trigger = {
							id = "close_planet"
							OR = {
								is_colonizable = yes
								is_colony = yes
							}
							NOT = {has_planet_flag = start_growth}
						}
						found_planet = {
							move_to = this
							effect = {
								id = "set_growth_flag"
								set_planet_flag = start_growth
							}
							wait = {duration = 45}
							effect = {
								id = "start_the_growth"
								set_variable = {which = army_count value = 0}
								planet_event = {id = fatalf.033}
								set_variable = { which = blocker_count value = 0 }
								planet_event = {id = fatalf.032}
							}
						}
					}
					find_closest_system = {
						trigger = {
							id = "system_search"
							OR = {
								any_planet = {
									OR = {
										is_colonizable = yes
										is_colony = yes
									}
									NOT = {has_planet_flag = start_growth}
								}
								exists = owner
							}
						}
						found_system = {
							move_to = this
						}
					}
				}
			}
		}
	}
}
				
#Start Ground Infection
planet_event = {
	id = fatalf.032
	hide_window = yes
	is_triggered_only = yes
	location = root
	
	immediate = {
		if = {						#AI colonies will only spread 5 times
			limit = {
				check_variable = { which = blocker_count value > 5 }
				exists = owner
				owner = {is_ai = yes}
			}
			break = yes
		}
		else_if = {											#If completely covered, clear a random space for reclaimation efforts and spawn new ship
			limit = {
				NOT = {
					any_tile = {
						has_blocker = no
					}
				}
			}
			random_tile = {
				remove_blocker = yes
			}
			if = {
				limit = {exists = event_target:overgrowth}
				event_target:overgrowth = {
					create_fleet = {
						name = "Seed Ship"
						effect = {
							set_owner = event_target:overgrowth
							create_ship = {
								name = random
								design = "Seed Ship"
							}
							set_location = {
								target = root
								distance = 10
								angle = random
							}
							queue_actions = {
								repeat = {
									find_closest_planet = {
										trigger = {
											id = "close_planet_2"
											OR = {
												is_colonizable = yes
												is_colony = yes
											}
											NOT = {has_planet_flag = start_growth}
											distance = {type = hyperlane}
										}
										found_planet = {
											move_to = this
											effect = {
												id = "set_growth_flag_2"
												set_planet_flag = start_growth
											}
											wait = {duration = 45}
											effect = {
												id = "start_the_growth_2"
												set_variable = {which = army_count value = 0}
												planet_event = {id = fatalf.033}
												set_variable = { which = blocker_count value = 0 }
												planet_event = {id = fatalf.032}
											}
										}
									}
									find_closest_system = {
										trigger = {
											id = "system_search_2"
											OR = {
												any_planet = {
													OR = {
														is_colonizable = yes
														is_colony = yes
													}
													NOT = {has_planet_flag = start_growth}
												}
												exists = owner
											}
										}
										found_system = {
											move_to = this
										}
									}
								}
							}
						}
					}
				}
			}
		}
		else_if = {										#If no starting infection, start the spread
			limit = {
				NOT = {
					any_tile = {
						has_blocker = tb_overgrowth
					}
				}
			}
			if = {										#If completely cleared and has spread 5 times, do not spread any more.
				limit = {check_variable = { which = blocker_count value > 5 }}
				planet_event = {id = fatalf.0305}
				break = yes
			}
			random_tile = {
				set_blocker = tb_overgrowth
			}
			root = {
				change_variable = { which = blocker_count value = 1 }
				planet_event = {id = fatalf.032 days = 45}
			}
		}
		else = {
			random_tile = {							#Continue the Overgrowth
				limit = {
					NOT = {has_blocker = tb_overgrowth}
					any_neighboring_tile = {
						has_blocker = tb_overgrowth
					}
				}
				set_blocker = tb_overgrowth
			}
			root = {
				create_army = {
					name = "Seed Army"
					owner = event_target:overgrowth
					species = event_target:overgrowth_species
					type = seed_army 
				}
				change_variable = { which = blocker_count value = 1 }
				planet_event = {id = fatalf.032 days = 45}
			}
		}
	}
}

#Spawn Spore Armies

	id = fatalf.033
	hide_window = yes
	is_triggered_only = yes
	location = root
	
	immediate = {
		if	= {
			limit = {
				is_colony = yes 
				exists = owner 
				owner = {
					NOT = {has_country_flag = overgrowth_ground_seen}
				}
			}
			planet_event = {id = fatalf.0304}
		}
		if = {
			limit = {is_colony = yes 	is_occupied_flag = no	check_variable = {which = army_count value <= 7 }}

			create_army = {
				name = "Seed Army"
				owner = event_target:overgrowth
				species = event_target:overgrowth_species

				name = "Seed Army"

				type = seed_army 
			}
			change_variable = {which = army_count  value = 1 }

		}
	}
}
