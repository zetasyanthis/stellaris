building_gms_xenophob = {
	base_buildtime = 2400
	empire_unique = yes
	
	cost = {
		minerals = 5000
		influence = 250
	}
	
	potential = {
		planet = {
			AND = {
				is_capital = yes
				#Empire Unique
				owner = {
					check_variable = {
						which = gms_government_built
						value = 0
					}
				}
				#Ethic
				owner = {
					OR = {
						has_ethic = ethic_xenophobe
						has_ethic = ethic_fanatic_xenophobe
					}
				}
			}
		}
	}
	
	allow = {
		custom_tooltip = {
			fail_text = "requires_any_empire_capital"
			planet = {
				NOR = {
					has_building = "building_capital_1"
					has_building = "building_capital_2"
				}		
			}
		}
	}
	
	destroy_if = {
		OR = {
			planet = {
				is_capital = no
			}
		}
	}
	
	on_queued = {
		planet = {
			owner = {
				if = {
					limit = {
						check_variable = {
							which = gms_government_built
							value < 1
						}
					}
					set_variable = {
						which = gms_government_built
						value = 1
					}
					
					else = {
						change_variable = {
							which = gms_government_built
							value = 1
						}
					}
				}
				set_country_flag = gms_on_tick_disabled
			}
		}
	}
	
	on_unqueued = {
		planet = {
			owner = {
				change_variable = {
					which = gms_government_built
					value = -1
				}
			}
		}
	}
	
	produced_resources = {
		influence = 0.5
		unity = 8
	}
	
	country_modifier = {
		country_core_sector_system_cap = 1
		country_government_civic_points_add = 1
		pop_government_ethic_attraction = 0.1
		max_rivalries = 1
		country_starbase_influence_cost_mult = -0.1
		rivalry_influence_gain = 0.25
		country_trust_growth = -0.1
		
	}
	
	prerequisites = {
		"ig_tech_government"
	}
	
	upgrades = {
		building_gms_f_xenophob
	}
}
