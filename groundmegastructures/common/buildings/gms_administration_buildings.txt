building_admincomplex_government_order = {
	base_buildtime = 400
	
	cost = {
		minerals = 500
		influence = 50
	}
	
	produced_resources = {
		influence = 0.25
		unity = 5
		society_research = 4
	}
	
	required_resources = {
		energy = 2
	}
	
	country_modifier = {
		edict_cost = -0.1
		edict_length_mult = 0.1
	}
	
	potential = {
		planet = {
			is_capital = yes
		}
	}
	
	destroy_if = {
		planet = { is_capital = no }
	}
	
	allow = {
		custom_tooltip = {
			fail_text = "requires_any_empire_capital"
			planet = {
				NOR = {
					has_building = "building_capital_1"
					has_building = "building_capital_2"
					has_building = "building_machine_capital_1"
					has_building = "building_machine_capital_2"
				}		
			}
		}
		custom_tooltip = {
			text = "can_be_built_5_times"
			planet = {
				owner = {
					check_variable = {
						which = ig_admincomplex_order
						value < 3
					}
				}
			}
		}
	}
	
	prerequisites = {
		"ig_tech_administration_complexes"
	}
	
	ai_weight = {
		weight = 100
	}
	
	on_queued = {
		planet = {
			owner = {
				if = {
					limit = {
						check_variable = {
							which = ig_admincomplex_order
							value < 1
						}
					}
					set_variable = {
						which = ig_admincomplex_order
						value = 1
					}
					
					else = {
						change_variable = {
							which = ig_admincomplex_order
							value = 1
						}
					}
				}
			}
		}
	}
	
	on_unqueued = {
		planet = {
			owner = {
				change_variable = {
					which = ig_admincomplex_order
					value = -1
				}
			}
		}
	}
}
